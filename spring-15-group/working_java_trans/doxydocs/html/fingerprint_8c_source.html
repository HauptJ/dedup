218 double get_similarity(chunks* data1, chunks* data2) {
219  int count1, count2, max;
220  count1 = data1->fingers;
221  count2 = data2->fingers;
222 
223  if (count1 > count2)
224  max = count1;
225  else
226  max = count2;
227 
228  if (count1 == 0 && count2 == 0)
229  return 1.0;
230 
231  return (double) (get_intersection_fingers(data1, data2) / max);

234 int get_intersection_fingers(chunks* data1, chunks* data2) {
235  int count = 0; // number of intersections
236  node* iter1 = data1->head->next;
237  node* iter2 = data2->head->next;
238 
239  // loop through both sets of fingerprints
240  while (iter1->next != NULL) {
241  while (iter2->next != NULL) {
242  if (iter1->data->hash == iter2->data->hash)
243  count++;
244  iter2 = iter2->next;
245  }
246  iter1 = iter1->next;
247  }
248  return count;
